S3

        
// 全ファイル確認
aws s3 ls ${s3バケットURL} --recursive
// ファイル検索
aws s3 ls ${s3バケットURL} --recursive | findstr ${検索キー}
// ファイル検索と書き出し
aws s3 ls ${s3バケットURL} --recursive | findstr ${検索キー} > hogehoge.csv
// ファイル削除
rm ${ファイル名}
// ファイルのダウンロード
aws s3 cp ${s3バケットURL} . --recursive
